5. CSV EXPORT VIEWS
Export Students CSV
@login_required
def export_students_csv(request):
    response = HttpResponse(content_type='text/csv')
    response['Content-Disposition'] = 'attachment; filename=students.csv'

    writer = csv.writer(response)
    writer.writerow(['ID', 'First Name', 'Last Name', 'Email', 'Date of Birth', 'Enrollment Date'])

    for student in Student.objects.all():
        writer.writerow([
            student.id,
            student.first_name,
            student.last_name,
            student.email,
            student.date_of_birth,
            student.enrollment_date,
        ])

    return response

Export Enrollments CSV
@login_required
def export_enrollments_csv(request):
    response = HttpResponse(content_type='text/csv')
    response['Content-Disposition'] = 'attachment; filename=enrollments.csv'

    writer = csv.writer(response)
    writer.writerow(['ID', 'Student', 'Course', 'Enrollment Date', 'Grade'])

    for enr in Enrollment.objects.select_related("student", "course"):
        writer.writerow([
            enr.id,
            f"{enr.student.first_name} {enr.student.last_name}",
            enr.course.name,
            enr.enrollment_date,
            enr.grade,
        ])

    return response
